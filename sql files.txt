CREATE  TABLE students_staff_map (
map_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
student_id INT NOT NULL,
staff_id INT NOT NULL,
FOREIGN KEY (student_id) REFERENCES students(id),
FOREIGN KEY (staff_id) REFERENCES staff_new(staff_id)
);

		st.name,
        st.address,
        sf.staff_id,
        sf.name AS staff_name,
        sf.designation
FROM students st 
JOIN students_staff_map m ON st.id = m.student_id 
JOIN staff_new sf ON m.staff_id = sf.staff_id;

SELECT st.id,st.name AS student_name,st.address,
	   sf.staff_id,sf.name AS staff_name,sf.designation
FROM students st 
LEFT JOIN students_staff_map m ON st.id = m.student_id

SELECT 
    YEAR(Admission_Date) AS year,
    MONTH(Admission_Date) AS month,
    COUNT(*) AS total_admissions,
    CASE 
        WHEN COUNT(*) >= 6600 THEN "PEAK MONTH"
        WHEN COUNT(*) >= 6300 THEN "AVERAGE MONTH"
        ELSE "LOW MONTH"
    END AS MOST_ADMISSONED_MONTH
FROM hospital_record
GROUP BY YEAR(Admission_Date), MONTH(Admission_Date)
ORDER BY year, month;

    st.id AS student_id,
    st.name AS student_name,
    st.address,

    sf.staff_id,
    sf.staff_name AS staff_name,
    sf.designation,

    -- CASE for conditional output
    CASE 
        WHEN sf.designation = 'Trainer' THEN 'Teaching Staff'
        WHEN sf.designation = 'Admin'   THEN 'Administrative Staff'
        ELSE 'Other Staff'
    END AS staff_category,

    -- Subquery: how many students this staff manages
    (
        SELECT COUNT(*)
        FROM students_staffs_map m2
        WHERE m2.staf_id = sf.staff_id
    ) AS total_students_under_staff,

    -- IF condition
    IF(
        sf.staff_id IS NULL, 
        'Not Assigned', 
        'Assigned'
    ) AS staff_status

FROM students st

-- LEFT JOIN so all students appear
LEFT JOIN students_staffs_map m ON st.id = m.stu_id
LEFT JOIN staffs sf        ON m.staf_id = sf.staff_id

-- Filtering using EXISTS (subquery)
WHERE EXISTS (
    SELECT 1
    FROM students_staffs_map m3
    WHERE m3.stu_id = st.id
)

-- Grouping for aggregated analysis
GROUP BY st.id, sf.staff_id

-- HAVING constraint (applied after grouping)
HAVING total_students_under_staff >= 1

-- Sorted report
ORDER BY student_name ASC

LIMIT 10;

SELECT o.order_id,
       c.customer_name,
	   o.total_amount,
       o.paid_amount,
case
       when o.total_amount = o.paid_amount then 'fully paid'
       when o.paid_amount= '0' then'not_paid'
       else 'partially paid'
       end as payment
from orders o
join customers c on o.customer_id=c.customer_id
having o.total_amount <> o.paid_amount;

SELECT 
    p.payment_mode,
    count(p.payment_id) as total_payments,
    sum(p.amount) as total_amount,
    CASE
        WHEN SUM(p.amount) > 200000 THEN 'High Usage'
        WHEN SUM(p.amount) BETWEEN 100000 AND 200000 THEN 'Medium Usage'
        ELSE 'Low Usage'
    END AS payment_rating
from payments p
group by p.payment_mode
HAVING sum(p.amount) > 0;

select 
    c.customer_name,
    (select count(*) 
     from orders o1 
     where o1.customer_id = c.customer_id
       and YEAR(o1.order_date) = YEAR(CURDATE()) - 1
    ) as last_year_orders,
    (select COUNT(*) 
     from orders o2 
     where o2.customer_id = c.customer_id
       AND YEAR(o2.order_date) = YEAR(CURDATE())
    ) as current_year_orders
from customers c
where c.customer_id in (
    select customer_id
    from orders
    where YEAR(order_date) = YEAR(CURDATE()) - 1
)
and c.customer_id IN (
    select customer_id
    from orders
    where YEAR(order_date) = YEAR(CURDATE())
);

-- rank()--
--fetch the top 2 employees in each dept earning themax salary --
SELECT *
FROM (
    SELECT e.*,
           RANK() OVER (PARTITION BY dept_name ORDER BY emp_salary DESC) AS rnk
    FROM employee e
) x
WHERE x.rnk <= 2;
-- dense rank--
SELECT *
FROM (
    SELECT e.*,
           RANK() OVER (PARTITION BY dept_name ORDER BY emp_salary DESC) AS rnk,
           DENSE_RANK() OVER (PARTITION BY dept_name ORDER BY emp_salary DESC) AS rnks
    FROM employee e
) x
